1function bLessThanTwoUniqueRows = queryIfOnlyTwoUniqueRows(X)
 2% Function that checks if an array has only two unique rows as this can
 3% cause failure of for example LDA
 4if size(X,1)==2
 5    bLessThanTwoUniqueRows = true;
 6    return
 7end
 8bEqualFirst = all(bsxfun(@eq,X,X(1,:)),2);
 9iFirstNotEqual = find(~bEqualFirst,1);
10if isempty(iFirstNotEqual)
11    bLessThanTwoUniqueRows = true;
12    return;
13end
14iToCheck = find(~bEqualFirst(2:end))+1;
15bNotUnique = all(bsxfun(@eq,X(iToCheck,:),X(iFirstNotEqual(1),:)),2);
16bLessThanTwoUniqueRows = all(bNotUnique);
17end
